<script>
  /* ─────────────────────────────────────
     Greeting  (time-aware)
  ───────────────────────────────────── */
  (function () {
    const h = new Date().getHours();
    let greeting = 'Good Morning';
    if (h >= 12 && h < 17) greeting = 'Good Afternoon';
    else if (h >= 17)      greeting = 'Good Evening';

    const greetEl = document.querySelector('.greeting-sub');
    if (greetEl) greetEl.textContent = greeting + ',\u00A0';
  })();

  /* ─────────────────────────────────────
     Booking data
  ───────────────────────────────────── */
  const todayBookings = [
    {
      id: 'B001',
      turf: 'The Multi-Sports Turf (7v7)',
      date: '16 Mar 2026',
      time: '6:00 – 7:00 AM',
      status: 'booked',
    },
    {
      id: 'B002',
      turf: 'Ultimate Powerplay Turf',
      date: '16 Mar 2026',
      time: '8:00 – 9:00 AM',
      status: 'booked',
    },
    {
      id: 'B003',
      turf: 'Battlefield Delhi Vaishali Nagar',
      date: '16 Mar 2026',
      time: '10:00 – 11:30 AM',
      status: 'payment-due',
    },
    {
      id: 'B004',
      turf: 'Kes Turf',
      date: '16 Mar 2026',
      time: '2:00 – 3:00 PM',
      status: 'booked',
    },
    {
      id: 'B005',
      turf: 'Strike Hill Side Turf',
      date: '16 Mar 2026',
      time: '5:00 – 6:30 PM',
      status: 'booked',
    },
  ];

  /* ─────────────────────────────────────
     Render booking cards
  ───────────────────────────────────── */
  function getBorderColor(status) {
    if (status === 'payment-due') return 'var(--warning)';
    if (status === 'cancelled')   return 'var(--error-red)';
    return 'var(--green)';
  }

  function renderBookings() {
    const list = document.getElementById('bookings-list');
    if (!list) return;

    if (todayBookings.length === 0) {
      list.innerHTML = `
        <div class="flex flex-col items-center justify-center py-16 gap-3 text-center">
          <svg width="48" height="48" viewBox="0 0 24 24" fill="none">
            <rect x="3" y="4" width="18" height="18" rx="2" stroke="var(--text-placeholder)" stroke-width="1.5"/>
            <path d="M16 2V6M8 2V6M3 10H21" stroke="var(--text-placeholder)" stroke-width="1.5" stroke-linecap="round"/>
          </svg>
          <p class="text-[var(--text-body)] text-sm">No bookings for today.</p>
        </div>`;
      return;
    }

    list.innerHTML = todayBookings.map(b => `
      <a href="Booking.html"
         class="bg-white rounded-[4px] flex items-center justify-between px-4 py-3 hover:shadow-sm transition-shadow"
         style="border-left: 5px solid ${getBorderColor(b.status)};"
         aria-label="Booking ${b.id}"
      >
        <div class="flex flex-col gap-1 min-w-0 flex-1 pr-3">
          <p class="text-[var(--text-heading)] text-[16px] font-semibold leading-7 truncate">${b.turf}</p>
          <p class="text-[var(--text-body)] text-[14px] font-normal leading-[21px] tracking-[0.04px]">${b.date}\u00A0\u2022\u00A0${b.time}</p>
        </div>
        <!-- Chevron right -->
        <svg width="20" height="20" viewBox="0 0 24 24" fill="none" flex-shrink="0">
          <path d="M9 18L15 12L9 6" stroke="var(--text-body)" stroke-width="1.5" stroke-linecap="round" stroke-linejoin="round"/>
        </svg>
      </a>
    `).join('');
  }

  renderBookings();

  /* ─────────────────────────────────────
     Banner carousel (auto-slide + dots)
  ───────────────────────────────────── */
  (function () {
    const track = document.getElementById('banner-track');
    const dots  = document.querySelectorAll('#banner-dots .dot');
    if (!track || !dots.length) return;

    const total = dots.length;
    let current = 0;
    let timer;

    function goTo(idx) {
      current = idx;
      track.style.transform = `translateX(-${current * 100}%)`;
      dots.forEach((d, i) => {
        d.classList.toggle('active', i === current);
      });
    }

    function next() { goTo((current + 1) % total); }

    function startTimer() {
      clearInterval(timer);
      timer = setInterval(next, 4000);
    }

    // Dot click
    dots.forEach(d => {
      d.addEventListener('click', () => {
        goTo(parseInt(d.dataset.idx));
        startTimer();
      });
    });

    // Swipe support
    let startX = 0;
    const wrapper = document.getElementById('banner-wrapper');
    wrapper.addEventListener('touchstart', e => { startX = e.touches[0].clientX; }, { passive: true });
    wrapper.addEventListener('touchend', e => {
      const dx = e.changedTouches[0].clientX - startX;
      if (Math.abs(dx) > 40) {
        goTo(dx < 0 ? (current + 1) % total : (current - 1 + total) % total);
        startTimer();
      }
    });

    startTimer();
  })();
</script>
